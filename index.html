yum安装mysql

#安装命令
yum install mysql-server

#后台启动
/etc/init.d/mysqld start &

#设置本地初始密码
grant all privileges on . to root@'localhost' identified by 'root' with grant option;

#删除匿名用户 delete from mysql.user where user = "";

#刷新权限
flush privileges;

说明：默认配置文件为 /etc/my.cnf
